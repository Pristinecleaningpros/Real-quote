    function calculateEstimate() {
      const sqft = parseInt(document.getElementById('sqft').value) || 0;
      const bedrooms = parseInt(document.getElementById('bedrooms').value) || 0;
      const bathrooms = parseInt(document.getElementById('bathrooms').value) || 0;
      const cleanType = document.getElementById('cleanType').value;
      const frequency = document.getElementById('frequency').value;
      const addons = Array.from(document.getElementById('addons').selectedOptions).map(opt => opt.value);

      let baseRate = 0.10; // $ per sqft
      if (cleanType === "deep") baseRate += 0.05;
      if (cleanType === "move") baseRate += 0.08;

      let total = sqft * baseRate + (bedrooms * 10) + (bathrooms * 15);

      // Frequency discounts
      if (frequency === "weekly") total *= 0.85;
      if (frequency === "biweekly") total *= 0.90;
      if (frequency === "monthly") total *= 0.95;

      // Add-ons
      const addonPrices = {
        fridge: 20,
        oven: 25,
        laundry: 15,
        windows: 30
      };
      addons.forEach(addon => {
        total += addonPrices[addon] || 0;
      });

      document.getElementById('estimateOutput').innerText = `Estimated Cost: $${total.toFixed(2)}`;
    }
  </script>
